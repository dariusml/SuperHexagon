184 void update_loading ()
185 {
186 Graphics_Context::Accessor context = director.lock_graphics_context ();
187
188 if (context)
189 {
190 //Se ajusta el aspect ratio, que servira para las demas escenas
191 adjustAspectRatio(context);
192
193 // Se carga la textura del logo:
194
195 logo_texture = Texture_2D::create (0, context, "logo.png");
196
197 // Se comprueba si la textura se ha podido cargar correctamente:
198
199 if (logo_texture)
200 {
201 context->add (logo_texture);
202
203 timer.reset ();
204
205 opacity = 0.f;
206 state = FADING_IN;
207 }

213 void update_fading_in ()
214 {
215 float elapsed_seconds = timer.get_elapsed_seconds ();
216
217 if (elapsed_seconds < 1.f)
218 {
219 opacity = elapsed_seconds; // Se aumenta la opacidad del logo a medida que pasa el tiempo

230 void update_waiting ()

233
234 if (timer.get_elapsed_seconds () > 2.f)
235 {
236 timer.reset ();
237
238 state = FADING_OUT;
239 }
240 }
241
242 void update_fading_out ()
243 {
244 float elapsed_seconds = timer.get_elapsed_seconds ();
245
246 if (elapsed_seconds < .5f)
247 {
248 opacity = 1.f - elapsed_seconds * 2.f; // Se reduce la opacidad de 1 a 0 en medio segundo
249 }
250 else
251 {
252 // Cuando el faceout se ha completado, se lanza la siguiente escena:
253
254 state = FINISHED;
255
256 director.run_scene (shared_ptr< Scene >(new Menu_Scene));
