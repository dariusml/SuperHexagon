269 //Saco los vertices de un hexagono, para poder hacer las secciones

272
273 vertices.push_back({xValue + xOffset, yValue + yOffset});
274 vertices.push_back({length + xOffset, yOffset});
275 vertices.push_back({xValue + xOffset, -yValue + yOffset});
276 vertices.push_back({-xValue + xOffset, -yValue + yOffset} );
277 vertices.push_back({-length + xOffset, yOffset} );
