144 void update(float deltaTime)
145 {
146 lastFrameDistanceFromCenterSqr = SqrDistanceFromCenter(vertices[0]);
147
148 float cosOfHalfAngle = cos(getAngleOfSection() * 0.5f);
149
150 //seteo la posicion proporcional respecto a la seccion de los vertices más adelantados del obstaculo
151
152 proportion -= (speed * deltaTime / cosOfHalfAngle) / parentSize * cosOfHalfAngle;
153
154 //seteo la anchura proporcional respecto a la seccion de los vertices más adelantados del obstaculo
155 widthProportion = width / cosOfHalfAngle / parentSize * cosOfHalfAngle;
156
157 //consigo el centro donde se juntan todas las secciones
158 Point2f centro = parentSection->get_vertice_position(0);
159
160 //consigo el centro donde se juntan todas las seccionescoloco los vertices del obstaculo
161 vertices[0] = makePointFromSegment(centro, parentSection->get_vertice_position(1), proportion);
162 vertices[1] = makePointFromSegment(centro, parentSection->get_vertice_position(1), proportion + widthProportion);
163 vertices[2] = makePointFromSegment(centro, parentSection->get_vertice_position(2), proportion + widthProportion);

170 //A partir de un segmento definido poe 2 puntos, escala el segmento en relacion al punto A
171 Point2f makePointFromSegment(const Point2f A, const Point2f B, float proportion)
172 {
173 Point2f finalPoint;
174 //Si ya ha llegado al centro, no continua
175 if(proportion > 0)
176 {
177 Vector2f vectorDiference{ B.coordinates.x() - A.coordinates.x(), B.coordinates.y() - A.coordinates.y() };
178
179 vectorDiference *= proportion;
180
181 finalPoint = { vectorDiference.coordinates.x() + A.coordinates.x(), vectorDiference.coordinates.y() + A.coordinates.y() };
182 } else{
183 finalPoint = {A.coordinates.x(), A.coordinates.y()};
184 }
185
186 return finalPoint;

191 float getAngleOfSection()
192 {
193 float dy;
194 float dx;
195
196 dy = parentSection->get_vertice_position(1).coordinates.y() - parentSection->get_vertice_position(0).coordinates.y();
197 dx = parentSection->get_vertice_position(1).coordinates.x() - parentSection->get_vertice_position(0).coordinates.x();
198
199 float angulo1 = atan2(dy,dx);
200
201 dy = parentSection->get_vertice_position(2).coordinates.y() - parentSection->get_vertice_position(0).coordinates.y();
202 dx = parentSection->get_vertice_position(2).coordinates.x() - parentSection->get_vertice_position(0).coordinates.x();
203
204 float angulo2 = atan2(dy,dx);
205
206 float anguloFinal = abs(angulo2 - angulo1) ;
207 return anguloFinal;
208 }

210 float SqrDistanceFromCenter(const Point2f & point)
211 {
212 float dxObstacle = point.coordinates.x() - parentSection->get_vertice_position(0).coordinates.x();
213 float dyObstacle = point.coordinates.y() - parentSection->get_vertice_position(0).coordinates.y();
214
215 return (dxObstacle * dxObstacle + dyObstacle * dyObstacle);
216 };

223 TypeOfCollision typeOfCollision(const Point2f & point)
224 {
225 TypeOfCollision typeOfCollision;
226
227 float result;
228 float currentSign = 0;
229 bool inside = true;
230 for(int i = 0; i < nVertices && inside; ++i)
231 {
232 result = (point.coordinates.y() - vertices[i].coordinates.y()) * (vertices[(i+1)%nVertices].coordinates.x() - vertices[i].coordinates.x()) - (point.coordinates.x() - vertices[i].coordinates.x()) * (vertices[(i+1)%nVertices].coordinates.y() -vertices[i].coordinates.y());
233 if(currentSign == 0 && result != 0)
234 {
235 currentSign = result;
236 }
237
238 //mira si tiene el mismo signo que el anterior
239 if( !( (currentSign >= 0 && result >= 0) || (currentSign <= 0 && result <= 0) ) ){
240 inside = false;
241 }
242
243 }
244
245 if(inside)
246 {
247 //Distancia que tiene el punto con el centro
248 float dxPoint = point.coordinates.x() - parentSection->get_vertice_position(0).coordinates.x();
249 float dyPoint = point.coordinates.y() - parentSection->get_vertice_position(0).coordinates.y();
250
251 float tipDistanceFromCenterSqr = dxPoint * dxPoint + dyPoint * dyPoint;
252
253 if(lastFrameDistanceFromCenterSqr <= tipDistanceFromCenterSqr) typeOfCollision = SIDECOLLISION;
254 else typeOfCollision = FRONTCOLLISION;
255
256 } else
257 {
258 typeOfCollision = NOCOLLISION;
259 }
260
261 return typeOfCollision;
